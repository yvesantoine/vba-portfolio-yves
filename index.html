    Sub CopyRowsBasedOnConcat()
    Dim sources As Variant
    sources = Array("Prilog D.9-S09", "Prilog D.10-S10", _
                   "Prilog D.11-S11", "Prilog D.12-S12")
    
    For Each src In sources
        Set ws = ThisWorkbook.Sheets(src)
        lastRow = ws.Cells(ws.Rows.Count, "G").End(xlUp).Row
        
        For i = 2 To lastRow
            If ws.Cells(i, 7).HasFormula And _
               InStr(1, ws.Cells(i, 7).Formula, "CONCAT") > 0 Then
                ' Process and copy data
            End If
        Next i
    Next src
End Sub
